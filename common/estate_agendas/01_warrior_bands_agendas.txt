estate_warrior_bands_hire_advisor = {
	can_select = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		employed_advisor = {
			category = MIL
		}
		is_in_deficit = no
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_warrior_bands
			loyalty = 10
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_warrior_bands
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

estate_warrior_bands_fire_advisor = {
	can_select = {
		employed_advisor = {
			category = MIL
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			always = yes
		}
		modifier = {
			factor = 0.5
			employed_advisor = {
				type = commandant
				religion = root
			}
		}
		modifier = {
			factor = 2
			advisor = commandant
			NOT = {
				employed_advisor = {
					type = commandant
					religion = root
				}
			}
		}
	}
	immediate_effect = {
		hidden_effect = {
			set_country_flag = need_to_fire_mil_advisor
		}
	}
	on_invalid = {
		clr_country_flag = need_to_fire_mil_advisor
	}
	task_requirements = {
		custom_trigger_tooltip = {
			tooltip = fire_mil_advisor_tooltip
			has_country_flag = just_fired_mil_advisor
		}
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_warrior_bands
			loyalty = 10
		}
		clr_country_flag = just_fired_mil_advisor
		clr_country_flag = need_to_fire_mil_advisor
	}
	invalid_trigger = {
		NOT = {
			employed_advisor = {
				category = MIL
			}
		}
		NOT = { has_country_flag = just_fired_mil_advisor }
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_warrior_bands
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		clr_country_flag = need_to_fire_mil_advisor
    }
}

#Recover Manpower
estate_warrior_bands_recover_manpower = {
	can_select = {
		manpower_percentage = 0.25
		is_at_war = no
		NOT = { has_country_modifier = recovering_economy }
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 1.5
			manpower_percentage = 0.15
		}
		modifier = {
			factor = 1.5
			manpower_percentage = 0.05
		}
	}
	fail_if = {
		is_at_war = yes
	}
	task_requirements = {
		manpower_percentage = 0.85 }
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_warrior_bands
			loyalty = 10
		}
		add_country_modifier = {
			name = recovering_economy
			duration = 5475
		}
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_warrior_bands
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
	}
}

estate_warrior_bands_build_barracks_in_y = {
	can_select = {
		mil_tech = 6
		any_owned_province = {
			is_state_core = root
			NOT = { has_building = barracks }
			OR = {
				AND = {
					NOT = { has_building = barracks }
					num_free_building_slots = 1
				}
				AND = {
					ROOT = { mil_tech = 16 }
					has_building = training_fields
				}
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.65
			NOT = {
				any_owned_province = {
					is_state_core = root
					base_manpower = 4
					NOT = { has_building = barracks }
					OR = {
						AND = {
							NOT = { has_building = barracks }
							num_free_building_slots = 1
						}
						AND = {
							ROOT = { mil_tech = 16 }
							has_building = training_fields
						}
					}
				}
			}
		}
		modifier = {
			factor = 1.2
			any_owned_province = {
				is_state_core = root
				base_manpower = 7
				NOT = { has_building = training_fields }
				OR = {
					AND = {
						NOT = { has_building = barracks }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { mil_tech = 16 }
						has_building = barracks
					}
				}
			}
		}
		modifier = {
			factor = 2
			NOT = {
				crown_land_share = 15
			}
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_warrior_bands_build_barracks_in_y_var
			value = 0
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { has_building = training_fields }
				OR = {
					AND = {
						NOT = { has_building = barracks }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { mil_tech = 16 }
						has_building = barracks
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				base_manpower = 4
				NOT = { has_building = training_fields }
				OR = {
					AND = {
						NOT = { has_building = barracks }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { mil_tech = 16 }
						has_building = barracks
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				base_manpower = 7
				NOT = { has_building = training_fields }
				OR = {
					AND = {
						NOT = { has_building = barracks }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { mil_tech = 16 }
						has_building = barracks
					}
				}
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			if = {
				limit = {
					NOT = { has_building = barracks }
				}
				ROOT = {
					set_variable = {
						which = estate_warrior_bands_build_barracks_in_y_var
						value = 1
					}
				}
			}
			else = {
				ROOT = {
					set_variable = {
						which = estate_warrior_bands_build_barracks_in_y_var
						value = 2
					}
				}
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		if = {
			limit = {
				check_variable = {
					which = estate_warrior_bands_build_barracks_in_y_var
					value = 2
				}
			}
			event_target:agenda_province = {
				has_building = training_fields
			}
		}
		else = {
			event_target:agenda_province = {
				OR = {
					has_building = barracks
					has_building = training_fields
				}
			}
		}
    }
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_warrior_bands
			loyalty = 10
		}
		if = {
			limit = {
				NOT = { crown_land_share = 15 }
				estate_territory = {
					estate = estate_warrior_bands
					territory = 10
				}
			}
			take_estate_land_share_small = { estate = estate_warrior_bands }
		}
		else = {
			event_target:agenda_province = {
				add_base_manpower = 1
			}
		}
    }
    failing_effect = {
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_warrior_bands_displeased
					value = 5
				}
			}
		}
		add_estate_loyalty_modifier = {
			estate = estate_warrior_bands
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
    }
}
