pw_is_althistorical_zunist = {
	custom_trigger_tooltip = {
		tooltip = pw_althistorical_zunist_tt		
		OR = {
			AND = {
				OR = {
					culture = baluchi
					culture = afghani
					culture = tajik
				}
				capital_scope = {
					OR = {
						area = ghor_area 
						area = balkh_area 
						area = kabulistan_area 
						area = makran_area 
						area = kalat_area 	
					}
				}
			}
		}
	}
}


pw_can_convert_to_zunist = {
	OR = {
		pw_is_althistorical_turkic_pagan = yes
		any_owned_province = {
			religion = zunist
		}
	}
	if = {
		limit = {
			capital_scope = {
				OR = {
					area = ghor_area 
					area = balkh_area 
					area = kabulistan_area 
					area = makran_area 
					area = kalat_area 	
				}
			}
		}
		OR = {
			num_of_religion = {
				value = 0.25
				religion = zunist
			}
			custom_trigger_tooltip = {
				tooltip = pw_can_convert_to_zunist_tt_1
				is_subject = no
				num_of_religion = {
					value = 0.15
					religion = zunist
				}
			}
			custom_trigger_tooltip = {
				tooltip = pw_can_convert_to_zunist_tt_2
				any_owned_province = {
					religion = zunist
					is_reformation_center = yes
				}
				num_of_religion = {
					value = 0.1
					religion = zunist
				}
			}
		}
	}
}
