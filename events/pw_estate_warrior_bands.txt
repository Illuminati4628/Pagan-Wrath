namespace = pw_estate_warrior_bands

country_event = { #Advancement of the Warrior Bands
	id = pw_estate_warrior_bands.1
	title = pw_estate_warrior_bands.1.t
	desc = pw_estate_warrior_bands.1.d
	picture = NOBLE_ESTATE_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		has_estate = estate_warrior_bands
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_warrior_bands
				modifier = EST_VAL_WARRIOR_BANDS_DECLINES
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_warrior_bands
				modifier = EST_VAL_WARRIOR_BANDS_ADVANCES
			}
		}
		NOT = {
			has_estate_influence_modifier = {
				estate = estate_warrior_bands
				modifier = EST_VAL_WARRIOR_BANDS_ADVANCES_20
			}
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	

	option = {
		name = pw_estate_warrior_bands.1.a #Ok
		add_estate_influence_modifier = {
			estate = estate_warrior_bands
			desc = EST_VAL_WARRIOR_BANDS_ADVANCES
			influence = 10
			duration = 5475
		}
	}
}
