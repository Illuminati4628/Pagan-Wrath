country_event = {
	id = pw_forceconversion.1
	title =  pw_forceconversion.1.t
	desc = pw_forceconversion.1.desc
	picture = RELIGION_eventPicture
	
	trigger = {
		ROOT = any_owned_province = { religion = norse_pagan_reformed }
		NOT = { religious_unity = 0.9 }
		religion_years = { norse_pagan_reformed = 5 }
		NOT = { religion = norse_pagan_reformed }
		culture_group = scandinavian
	}
	mean_time_to_happen = {
		months = 1200
	}
	option = { #Perhaps this is the best way foward
		name = pw_forceconversion.1.a
		change_religion = norse_pagan_reformed
		add_prestige = -50
		add_country_modifier = {
			name = "pagan_conversion_zeal"
			duration = 3650
		}
		ai_chance = {
			factor = 30
			modifier = {
				factor = 70
				num_of_religion = {
					religion = norse_pagan_reformed
					value = 0.3
				}	
			}
			modifier = {
				factor = 0
				OR = {
					ruler_has_personality = tolerant_personality
					ruler_has_personality = zealot_personality
				}	
			}			
		}
	}
	option = {
		name = pw_forceconversion.1.b #We will not allow it
		add_prestige = 20
		add_country_modifier = {
			name = "pw_religious_violance"
			duration = 3650
		}
		ai_chance = {
			factor = 70
			modifier = {
				factor = 30
				NOT = {
					num_of_religion = {
						religion = norse_pagan_reformed
						value = 0.3
					}	
				}	
			}
			modifier = {
				factor = 0
				OR = {
					ruler_has_personality = tolerant_personality
					ruler_has_personality = zealot_personality
				}	
			}			
		}
	}	
	option = {
		name = pw_forceconversion.1.c #Perhaps we can come to a compromise?
		highlight = yes
		trigger = {
			ruler_has_personality = tolerant_personality
		}
		required_personality = tolerant_personality
		add_prestige = -20
		add_country_modifier = {
			name = "pw_religious_tolerance"
			duration = 3650
		}
		ai_chance = {
			factor = 100
		}	
	}
	option = {
		name = pw_forceconversion.1.d #Deal with these heathens at once!
		highlight = yes
		trigger = {
			ruler_has_personality = zealot_personality
		}
		required_personality = zealot_personality		
		add_prestige = 25
		add_country_modifier = {
			name = "pw_anti_pagan_conversion"
			duration = 3650
		}
		ai_chance = {
			factor = 100
		}		
	}
}
