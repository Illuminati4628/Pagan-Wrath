namespace = pw_germanic_tournament

country_event = { # 1 The Championship of the Warriors
	id = pw_germanic_tournament.1
	title = "pw_germanic_tournament.1.t"
	desc = "pw_germanic_tournament.1.d"
	picture = TOURNAMENT_eventPicture
	
	major = no
	
	trigger = {
		has_country_flag = warrior_bands_tournaments
		religion_group = germanic_group
		OR = {
			NOT = { has_country_flag = germanic_had_recent_tournament }
			had_country_flag = { flag = germanic_had_recent_tournament days = 1825 }
		}		
	}	
	
	immediate = {
		hidden_effect = {
			set_country_flag = germanic_had_recent_tournament
		}
	}
	
	mean_time_to_happen = {
		months = 2
	}	
	
	option = { #Let the championship begin!
		name = "pw_germanic_tournament.1.a"
		ai_chance = { factor = 1 }
		
		add_years_of_income = -0.1
		add_legitimacy = 5
		germanic_tournament_effect = yes
	}
}
